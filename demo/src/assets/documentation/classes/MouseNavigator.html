import { AbstractNavigator, NavigationAction} from './navigator';


/**
 * An implementation of [AbstractNavigator]{@link AbstractNavigator} that is responsible for exitting
 * the Table Editor when clicking outside of the `HTMLTableElement`.
 */
export class MouseNavigator extends AbstractNavigator {
	public listener(): () => void {
		return this.renderer.listen(document, 'click', this.actionGenerator.bind(this));
	}
	private actionGenerator(event: MouseEvent): void {
		if (!(event as any)['path'].includes(this.tableElementRef.nativeElement)) {
			this.execute(NavigationAction.Exit);
		}
	}
}
